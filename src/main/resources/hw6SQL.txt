CREATE TABLE homework6.products (id serial, title varchar(100), price int, PRIMARY KEY (id));
CREATE TABLE homework6.customers (id serial, name varchar(100), PRIMARY KEY (id));
CREATE TABLE homework6.order_details (id serial, product_id bigint NOT NULL REFERENCES homework6.products(id), customer_id bigint NOT NULL REFERENCES homework6.customers(id), PRIMARY KEY (id));
